339 int8_t bme680_set_regs(const uint8_t *reg_addr, const uint8_t *reg_data, uint8_t len, struct bme680_dev *dev)
340 {
341  int8_t rslt;
342  /* Length of the temporary buffer is 2*(length of register)*/
343  uint8_t tmp_buff[BME680_TMP_BUFFER_LENGTH] = { 0 };
344  uint16_t index;
345 
346  /* Check for null pointer in the device structure*/
347  rslt = null_ptr_check(dev);
348  if (rslt == BME680_OK) {
349  if ((len > 0) && (len < BME680_TMP_BUFFER_LENGTH / 2)) {
350  /* Interleave the 2 arrays */
351  for (index = 0; index < len; index++) {
352  if (dev->intf == BME680_SPI_INTF) {
353  /* Set the memory page */
354  rslt = set_mem_page(reg_addr[index], dev);
355  tmp_buff[(2 * index)] = reg_addr[index] & BME680_SPI_WR_MSK;
356  } else {
357  tmp_buff[(2 * index)] = reg_addr[index];
358  }
359  tmp_buff[(2 * index) + 1] = reg_data[index];
360  }
361  /* Write the interleaved array */
362  if (rslt == BME680_OK) {
363  dev->com_rslt = dev->write(dev->dev_id, tmp_buff[0], &tmp_buff[1], (2 * len) - 1);
364  if (dev->com_rslt != 0)
365  rslt = BME680_E_COM_FAIL;
366  }
367  } else {
368  rslt = BME680_E_INVALID_LENGTH;
369  }
370  }
371 
372  return rslt;
373 }

577 {
578  int8_t rslt;
579  uint8_t tmp_pow_mode;
580  uint8_t pow_mode = 0;
581  uint8_t reg_addr = BME680_CONF_T_P_MODE_ADDR;
582 
583  /* Check for null pointer in the device structure*/
584  rslt = null_ptr_check(dev);
585  if (rslt == BME680_OK) {
586  /* Call repeatedly until in sleep */
587  do {
588  rslt = bme680_get_regs(BME680_CONF_T_P_MODE_ADDR, &tmp_pow_mode, 1, dev);
589  if (rslt == BME680_OK) {
590  /* Put to sleep before changing mode */
591  pow_mode = (tmp_pow_mode & BME680_MODE_MSK);
592 
593  if (pow_mode != BME680_SLEEP_MODE) {
594  tmp_pow_mode = tmp_pow_mode & (~BME680_MODE_MSK); /* Set to sleep */
595  rslt = bme680_set_regs(&reg_addr, &tmp_pow_mode, 1, dev);
596  dev->delay_ms(BME680_POLL_PERIOD_MS);
597  }
598  }
599  } while (pow_mode != BME680_SLEEP_MODE);
600 
601  /* Already in sleep */
602  if (dev->power_mode != BME680_SLEEP_MODE) {
603  tmp_pow_mode = (tmp_pow_mode & ~BME680_MODE_MSK) | (dev->power_mode & BME680_MODE_MSK);
604  if (rslt == BME680_OK)
605  rslt = bme680_set_regs(&reg_addr, &tmp_pow_mode, 1, dev);
606  }
607  }
608 
609  return rslt;
610 }

630 void bme680_set_profile_dur(uint16_t duration, struct bme680_dev *dev)
631 {
632  uint32_t tph_dur; /* Calculate in us */
633  uint32_t meas_cycles;
634  uint8_t os_to_meas_cycles[6] = {0, 1, 2, 4, 8, 16};
635 
636  meas_cycles = os_to_meas_cycles[dev->tph_sett.os_temp];
637  meas_cycles += os_to_meas_cycles[dev->tph_sett.os_pres];
638  meas_cycles += os_to_meas_cycles[dev->tph_sett.os_hum];
639 
640  /* TPH measurement duration */
641  tph_dur = meas_cycles * UINT32_C(1963);
642  tph_dur += UINT32_C(477 * 4); /* TPH switching duration */
643  tph_dur += UINT32_C(477 * 5); /* Gas measurement duration */
644  tph_dur += UINT32_C(500); /* Get it to the closest whole number.*/
645  tph_dur /= UINT32_C(1000); /* Convert to ms */
646 
647  tph_dur += UINT32_C(1); /* Wake up duration of 1ms */
648  /* The remaining time should be used for heating */
649  dev->gas_sett.heatr_dur = duration - (uint16_t) tph_dur;
650 }
